190Bundle split_args(string s) {
191 const char *spl = string(" ").c_str();
192 char *str = new char[90];
193 std::strcpy(str, s.c_str());
194 unsigned int count = count_char(s.c_str(), s.length(), ' ') + 1;
195 string *args = new string[count];
196 char *tok = std::strtok(str, spl);
197 int c = 0;
198 while (tok != NULL) {
199 args[c++] = string(tok);
200 tok = std::strtok(NULL, spl);
201 }
202 delete[] str;
203 return Bundle { args, count };
204}
